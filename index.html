// JavaScript
function convertTemp(value, from) {
  let c, f, k;
  if (from === "F") { c=(value-32)*5/9; f=value; k=c+273.15; }
  else if (from === "C") { c=value; f=(value*9/5)+32; k=value+273.15; }
  else { k=value; c=k-273.15; f=(c*9/5)+32; }
  console.log(f.toFixed(1)+"°F  "+c.toFixed(1)+"°C  "+k.toFixed(1)+"K");
}
convertTemp(98.6, "F");

// JavaScript
function gradeAverage(scores) {
  let total = 0;
  for (let i=0; i < scores.length; i++) { total += scores[i]; }

  let grade = avg>=90?"A":avg>=80?"B":avg>=70?"C":avg>=60?"D":"F";
  console.log(avg.toFixed(1) + " — " + grade);
}

import math

// JavaScript
function makeTeams(names, numTeams) {
  let shuffled = [...names].sort(() => Math.random() - 0.5);
  let teams    = Array.from({length:numTeams}, () => []);
  for (let i=0; i < shuffled.length; i++) teams[i%numTeams].push(shuffled[i]);
  teams.forEach((t,i) => console.log("Team "+(i+1)+": "+t.join(", ")));
}

# Python 3 — correct version
import random
def make_teams(names, num_teams):
    shuffled = names[:]
    random.shuffle(shuffled)
    teams = [[] for _ in range(num_teams)]
    for i, name in enumerate(shuffled):
        teams[i % num_teams].append(name)
    for i, team in enumerate(teams):
        print("Team " + str(i+1) + ": " + ", ".join(team))

// JavaScript
function checkPassword(pw) {
  let score = 0;
  if (pw.length >= 8)           score++;
  if (/[A-Z]/.test(pw))          score++;
  if (/[0-9]/.test(pw))          score++;
  if (/[^A-Za-z0-9]/.test(pw))   score++;
  if (pw.length >= 12)          score++;
  let labels=["Very weak","Weak","Okay","Strong","Very strong"];
  console.log(labels[score-1]||"Very weak");
}
checkPassword("Hello@123!");
# Python 3 — correct version
import re
def check_password(pw):
    score = 0
    if len(pw) >= 8:                    score += 1
    if re.search(r'[A-Z]', pw):         score += 1
    if re.search(r'[0-9]', pw):         score += 1
    if re.search(r'[^A-Za-z0-9]', pw):  score += 1
    if len(pw) >= 12:                   score += 1
    labels = ["Very weak","Weak","Okay","Strong","Very strong"]
    print(labels[score-1] if score>0 else "Very weak")
check_password("Hello@123!")

// JavaScript
function startTimer(minutes, seconds) {
  let remaining = minutes * 60 + seconds;
  let interval  = setInterval(() => {
    remaining--;
    let m=Math.floor(remaining/60), s=remaining%60;
    console.log(String(m).padStart(2,"0")+":"+String(s).padStart(2,"0"));
    if (remaining <= 0) { clearInterval(interval); console.log("Time's up!"); }
  }, 1000);
}
startTimer(0, 10);
# Python 3 — correct version
import time
def start_timer(minutes, seconds):
    remaining = minutes * 60 + seconds
    while remaining > 0:
        m = remaining // 60
        s = remaining % 60
        print(str(m).zfill(2) + ":" + str(s).zfill(2))
        time.sleep(1)
        remaining -= 1
    print("00:00 -- Time's up!")
start_timer(0, 5)
